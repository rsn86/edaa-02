\begin{algorithm}
    \caption{ - Busca sequencial \cite[p. 396]{knuth1998art}}
    \begin{algorithmic}[1]
        \STATE $i \leftarrow 1$
        \WHILE{$ i \leq n $}
            \IF{$ x = A[i]$}
                \RETURN $ i $
            \ENDIF
            \STATE $i \leftarrow i + 1$
        \ENDWHILE
        \RETURN \FALSE
    \end{algorithmic}
    \label{alg:busca_sequencial}
\end{algorithm}
